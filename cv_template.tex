% Dynamic sections populated from Notion database via Jinja2 templating
% These sections are generated automatically from the Notion database entries
% where "Show on CV?" is set to True. The sections include:
% - Work Experience: Professional work experience
% - Leadership and Other Experience: Volunteer work, clubs, activities, leadership roles
% - Projects: Personal and academic projects
% 
% Jinja2 template logic:
% - Loops through each section type
% - Filters entries where is_visible is True
% - Renders each entry with organization, location, dates, and description
% - Uses different formatting for Projects (no organization/role) vs others
{% for sect in ['Work Experience','Leadership and Other Experience','Projects'] %}
{% set items = cv_data.get(sect, [])|selectattr('is_visible')|list %}
{% if items|length %}
\section{ {{- sect | upper | latex_escape -}} }
{% for item in items %}
{% if sect == 'Projects' %}
% Projects section: Shows only project name and location (no organization/role)
% This format is cleaner for projects where the organization field might contain
% irrelevant role information like "Developer" or "Creator"
\begin{tabularx}{\linewidth}{@{}X r@{}}
\begin{minipage}[t]{\linewidth}
  \textbf{ {{- item.name | latex_escape -}} }
  {% if item.location %} -- {{ item.location | latex_escape }}{% endif %}
\end{minipage}
& {% if item.date_display %}
    {{ item.date_display | latex_escape }}
  {% elif item.start_date %}
    {{ (item.start_date ~ ' -- ' ~ (item.end_date or 'Present')) | latex_escape }}
  {% else %}
    {{ (item.end_date or '') | latex_escape }}
  {% endif %}
\end{tabularx}
{% else %}
% Work Experience and Leadership and Other Experience: Shows organization, location, and role/title
% This format works well for traditional work experience and activities
% where the organization and role are both relevant
\begin{tabularx}{\linewidth}{@{}X r@{}}
\begin{minipage}[t]{\linewidth}
  \textbf{ {{- item.organization | latex_escape -}} }
  {% if item.location %} -- {{ item.location | latex_escape }}{% endif %} \\
  \textbf{ {{- item.name | latex_escape -}} }
\end{minipage}
& {% if item.date_display %}
    {{ item.date_display | latex_escape }}
  {% elif item.start_date %}
    {{ (item.start_date ~ ' -- ' ~ (item.end_date or 'Present')) | latex_escape }}
  {% else %}
    {{ (item.end_date or '') | latex_escape }}
  {% endif %}
\end{tabularx}
{% endif %}
% Bullet points describing the role/project details
% Content comes from the "For Resume" section of each Notion page
% and is pre-processed to convert Notion formatting to LaTeX
\begin{itemize}[nosep,after=\strut, leftmargin=1em, itemsep=3pt,label=--]
  {{ item.body_latex | safe }}
\end{itemize}
{% endfor %}
{% endif %}
{% endfor %}
